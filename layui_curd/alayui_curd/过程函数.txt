名字随机
//通过工具创建不需要这一行CREATE FUNCTION `randStr`(n INT) RETURNS varchar(255) CHARSET utf8
BEGIN 
DECLARE chars_str varchar(100) DEFAULT '张王李赵钱徽音膺彦誉定干企禋雍安民治国吉祥安康兴旺发达丗丘丙业丛东丝丞丟丠両丢丣两严並'; 
DECLARE return_str varchar(255) DEFAULT '' ;
DECLARE i INT DEFAULT 0; 
WHILE i < n DO 
SET return_str = concat(return_str,substring(chars_str , FLOOR(1 + RAND()*39 ),1)); 
SET i = i +1; 
END WHILE; 
RETURN return_str; 
END
